% Problem Statement
\begin{block}{Problem Statement}
\textbf{The Core Question:}
How do Machine Learning and Statistical models compare across different problem domains in terms of:
\begin{itemize}
    \item \textcolor{mlcolor}{\textbf{Performance}} (accuracy, R², RMSE)
    \item \textcolor{statcolor}{\textbf{Cost}} (training time, computational resources)
    \item \textcolor{accentcolor}{\textbf{Interpretability}} (explainability, statistical inference)
\end{itemize}

\vspace{0.3cm}
\textbf{Research Objectives:}
\begin{enumerate}
    \item Compare ML and Statistical approaches across multiple problem domains
    \item Quantify trade-offs between performance, cost, and interpretability
    \item Provide evidence-based recommendations for model selection
    \item Connect findings to Interpretable ML research and Transparent AI
\end{enumerate}
\end{block}

% The Trade-offs
\begin{alertblock}{The Fundamental Trade-offs}
\begin{columns}[t]
\column{0.33\linewidth}
\begin{block}{\textcolor{mlcolor}{Performance}}
\begin{align*}
P(\theta) &= \mathbb{E}[\mathcal{M}(Y, \hat{Y})]
\end{align*}
Metrics: $R^2$, Accuracy, RMSE
\end{block}

\column{0.33\linewidth}
\begin{block}{\textcolor{statcolor}{Cost}}
\begin{align*}
C(\theta) &= T_{\text{train}}(\theta) + \lambda M(\theta)
\end{align*}
Training time + memory cost
\end{block}

\column{0.33\linewidth}
\begin{block}{\textcolor{accentcolor}{Interpretability}}
\begin{align*}
I(\theta) &= \text{Explainability Score} \in [0, 100]
\end{align*}
Transparency and trust
\end{block}
\end{columns}

\vspace{0.3cm}
\textbf{The Optimization Problem:}
\begin{align*}
\max_\theta \quad &P(\theta) - \alpha C(\theta) + \beta I(\theta) \\
\text{s.t. } &\text{constraints}
\end{align*}
\end{alertblock}

% When to Choose Statistical
\begin{block}{When to Choose Statistical Models}
\textcolor{statcolor}{\textbf{Choose Statistical Models When:}}
\begin{itemize}
    \item \textcolor{statcolor}{\ding{51}} Interpretability is critical - Need p-values, confidence intervals
    \item \textcolor{statcolor}{\ding{51}} Regulatory compliance - Transparent decision-making required
    \item \textcolor{statcolor}{\ding{51}} Hypothesis testing - Understanding causal relationships
    \item \textcolor{statcolor}{\ding{51}} Small to medium datasets - Optimal performance on limited data
    \item \textcolor{statcolor}{\ding{51}} Real-time applications - Fast training and prediction needed
    \item \textcolor{statcolor}{\ding{51}} Research and development - Need to understand underlying relationships
\end{itemize}

\vspace{0.3cm}
\textbf{Best Models:} OLS Regression, Logistic Regression, ANOVA, ARIMA, Cox Regression
\end{block}

% When to Choose ML/DL
\begin{block}{When to Choose ML/DL Models}
\textcolor{mlcolor}{\textbf{Choose ML/DL Models When:}}
\begin{itemize}
    \item \textcolor{mlcolor}{\ding{51}} Maximum accuracy is priority - Need best predictive performance
    \item \textcolor{mlcolor}{\ding{51}} Large, complex datasets - High-dimensional or non-linear patterns
    \item \textcolor{mlcolor}{\ding{51}} Production systems - Computational resources available
    \item \textcolor{mlcolor}{\ding{51}} Pattern discovery - Exploratory analysis on complex data
    \item \textcolor{mlcolor}{\ding{51}} Non-linear relationships - When linear assumptions fail
    \item \textcolor{dlcolor}{\ding{51}} Sequential/temporal data - LSTM/RNN for time series with complex patterns
\end{itemize}

\vspace{0.3cm}
\textbf{Best Models:} Random Forest, SVM, LSTM/RNN, Gradient Boosting
\end{block}

% Key Takeaways
\begin{block}{Key Takeaways}
\begin{enumerate}
    \item \textbf{No universal winner} - Both approaches have distinct advantages
    
    \item \textbf{Statistical models excel in:}
    \begin{itemize}
        \item Interpretability (88.3/100 vs 37.4/100)
        \item Speed (20-100x faster training)
        \item Efficiency (6435.64 vs 0.36 efficiency scores)
        \item Regulatory compliance and hypothesis testing
    \end{itemize}
    
    \item \textbf{ML/DL models excel in:}
    \begin{itemize}
        \item Complex dataset performance (0.818 vs 0.547 avg)
        \item Scalability to large data
        \item Capturing non-linear patterns
        \item Handling complex temporal dependencies
    \end{itemize}
    
    \item \textbf{The optimal choice depends on:}
    \begin{itemize}
        \item Primary requirement (accuracy vs interpretability vs speed)
        \item Dataset characteristics (size, complexity, linearity)
        \item Computational constraints
        \item Regulatory requirements
    \end{itemize}
\end{enumerate}
\end{block}
