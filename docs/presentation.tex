\section{Introduction}

\begin{frame}{Problem Statement}
\begin{block}{The Core Question}
How do Machine Learning and Statistical models compare across different problem domains in terms of:
\begin{itemize}
    \item \textcolor{mlcolor}{\textbf{Performance}} (accuracy, R², RMSE)
    \item \textcolor{statcolor}{\textbf{Cost}} (training time, computational resources)
    \item \textcolor{accentcolor}{\textbf{Interpretability}} (explainability, statistical inference)
\end{itemize}
\end{block}

\vspace{0.3cm}

\begin{block}{Research Objectives}
\begin{enumerate}
    \item Compare ML and Statistical approaches across multiple problem domains
    \item Quantify trade-offs between performance, cost, and interpretability
    \item Provide evidence-based recommendations for model selection
    \item Connect findings to Interpretable ML research and Transparent AI
\end{enumerate}
\end{block}

\subsection{Regression Analysis}

\begin{frame}{Regression: Models \& Datasets}
\begin{columns}
\column{0.5\textwidth}
\textcolor{statcolor}{\textbf{Statistical Model:}}
\begin{itemize}
    \item OLS Regression (statsmodels)
    \begin{align*}
    Y &= X\beta + \varepsilon \\
    \hat{\beta} &= (X^T X)^{-1} X^T Y
    \end{align*}
    \item Provides p-values, confidence intervals
    \item Statistical inference capabilities
\end{itemize}

\column{0.5\textwidth}
\textcolor{mlcolor}{\textbf{ML Model:}}
\begin{itemize}
    \item Linear Regression (scikit-learn)
    \begin{align*}
    \hat{y} &= f_\theta(X) \\
    \theta^* &= \arg\min_\theta \mathcal{L}(Y, f_\theta(X))
    \end{align*}
    \item Optimized for prediction
    \item Fast training and inference
\end{itemize}
\end{columns}

\vspace{0.5cm}

\begin{block}{Datasets}
\begin{itemize}
    \item \textbf{California Housing:} 20,640 samples, 8 features
    \item \textbf{Diabetes:} 442 samples, 10 features
    \item \textbf{Wine Quality:} 1,599 samples, 11 features
    \item \textbf{Energy Consumption:} 1,000 samples (synthetic), 6 features
\end{itemize}
\end{block}
\end{frame}

\begin{block}{Preprocessing Steps}

    \item Feature scaling for numerical stability

\subsection{Factorial Analysis: ANOVA vs SVM}

\begin{frame}{Factorial Analysis: Models \& Datasets}
\begin{columns}
\column{0.5\textwidth}
\textcolor{statcolor}{\textbf{Statistical Model:}}
\begin{itemize}
    \item ANOVA (Analysis of Variance)
    \begin{align*}
    Y_{ij} &= \mu + \alpha_i + \beta_j + (\alpha\beta)_{ij} + \varepsilon_{ij} \\
    F &= \frac{\text{MS}_{\text{between}}}{\text{MS}_{\text{within}}}
    \end{align*}
    \item Tests null hypothesis: $H_0: \mu_1 = \mu_2 = \ldots = \mu_k$
    \item Provides F-statistics and p-values
\end{itemize}

\column{0.5\textwidth}
\textcolor{mlcolor}{\textbf{ML Model:}}
\begin{itemize}
    \item Support Vector Regression (SVM)
    \begin{align*}
    \min_{\mathbf{w},b} &\quad \frac{1}{2}\|\mathbf{w}\|^2 + C\sum_{i=1}^n \xi_i \\
    K(\mathbf{x}_i, \mathbf{x}_j) &= \phi(\mathbf{x}_i)^T\phi(\mathbf{x}_j)
    \end{align*}
    \item RBF kernel for non-linear relationships
    \item Hyperparameter tuning via grid search
\end{itemize}
\end{columns}

\vspace{0.3cm}

\begin{block}{Datasets}
\begin{itemize}
    \item \textbf{Treatment Effects:} Clinical trial data
    \item \textbf{Educational Outcomes:} Teaching method comparison
    \item \textbf{Agricultural Experiment:} Fertilizer × Water level (3×3 factorial)
    \item \textbf{Drug Efficacy Study:} Drug treatment comparison
\end{itemize}
\end{block}
\end{frame}

\begin{frame}{Factorial Analysis: EDA \& Preprocessing}
\begin{block}{Exploratory Data Analysis}
\begin{itemize}
    \item Group means and standard deviations
    \item Box plots by treatment groups
    \item Interaction plots (for factorial designs)
    \item Assumption checking (normality, equal variances)
    \item Residual analysis
\end{itemize}
\end{block}

\vspace{0.3cm}

\begin{block}{Preprocessing}
\begin{itemize}
    \item Categorical encoding for group factors
    \item Feature engineering (interaction terms for ANOVA)
    \item Standardization for SVM
    \item Train-test split (80-20)
    \item Hyperparameter tuning for SVM (Grid Search CV)
\end{itemize}
\end{block}

\begin{frame}{Classification: Models \& Datasets}
\begin{columns}
\column{0.5\textwidth}
\textcolor{statcolor}{\textbf{Statistical Model:}}
\begin{itemize}
    \item Logistic Regression (statsmodels)
    \begin{align*}
    P(Y=1|X) &= \frac{1}{1 + e^{-X\beta}} \\
    \log\left(\frac{p}{1-p}\right) &= X\beta
    \end{align*}
    \item Maximum Likelihood Estimation
    \item Provides odds ratios, p-values
\end{itemize}

\column{0.5\textwidth}
\textcolor{mlcolor}{\textbf{ML Model:}}
\begin{itemize}
    \item Random Forest (scikit-learn)
    \begin{align*}
    \hat{y} &= \frac{1}{B}\sum_{b=1}^B T_b(X; \theta_b) \\
    T_b &\sim \text{Decision Tree}
    \end{align*}
    \item Ensemble of $B$ trees
    \item Feature importance scores
\end{itemize}
\end{columns}

\begin{itemize}
    \item ARIMA (AutoRegressive Integrated Moving Average)
    \begin{align*}
    \phi(B)(1-B)^d X_t &= \theta(B)\varepsilon_t \\

    \item Auto-selection of $(p,d,q)$

    \item Captures long-term dependencies

\section{The Trade-offs}

\begin{frame}{The Fundamental Trade-offs}
% \begin{center}
% \Large\textbf{Three Critical Dimensions}
% \end{center}

\vspace{0.0cm}

\begin{columns}
\column{0.33\textwidth}
\begin{block}{\textcolor{mlcolor}{Performance}}
\begin{align*}
P(\theta) &= \mathbb{E}[\mathcal{M}(Y, \hat{Y})] \\
&= \frac{1}{n}\sum_{i=1}^n \mathcal{M}(y_i, f_\theta(x_i))
\end{align*}
Metrics: $R^2$, Accuracy, RMSE
\end{block}

\column{0.33\textwidth}
\begin{block}{\textcolor{statcolor}{Cost}}
\begin{align*}
C(\theta) &= T_{\text{train}}(\theta) + \lambda M(\theta) \\
&\text{Training time} + \text{memory cost}
\end{align*}
Computational complexity
\end{block}

\column{0.33\textwidth}
\begin{block}{\textcolor{accentcolor}{Interpretability}}
\begin{align*}
I(\theta) &= \text{Explainability Score} \\
&\in [0, 100]
\end{align*}
Transparency and trust
\end{block}
\end{columns}

\vspace{0.0cm}

\begin{alertblock}{Key Insight}
The optimization problem:
\begin{align*}
\max_\theta \quad &P(\theta) - \alpha C(\theta) + \beta I(\theta) \\
\text{s.t. } &\text{constraints}
\end{align*}
% No single model excels in all three dimensions.
\end{alertblock}
\end{frame}

\section{Conclusions}

\begin{frame}{When to Choose Statistical Models}
\begin{block}{\textcolor{statcolor}{Choose Statistical Models When:}}
\begin{itemize}
\item \textcolor{statcolor}{\textbf{Interpretability is critical}} - Need p-values, confidence intervals, and statistical inference
\item \textcolor{statcolor}{\textbf{Regulatory compliance}} - Transparent decision-making required (e.g., healthcare, finance)
\item \textcolor{statcolor}{\textbf{Hypothesis testing}} - Understanding causal relationships and testing theories
\item \textcolor{statcolor}{\textbf{Small to medium datasets}} - Optimal performance on limited data
\item \textcolor{statcolor}{\textbf{Real-time applications}} - Fast training and prediction needed
\item \textcolor{statcolor}{\textbf{Research and development}} - Need to understand underlying relationships
\end{itemize}
\end{block}

\vspace{0.3cm}

\begin{center}
\textbf{Best Models:} OLS Regression, Logistic Regression, ANOVA, ARIMA, Cox Regression
\end{center}
\end{frame}

\begin{frame}{When to Choose ML/DL Models}
\begin{block}{\textcolor{mlcolor}{Choose ML/DL Models When:}}
\begin{itemize}
\item \textcolor{mlcolor}{\textbf{Maximum accuracy is priority}} - Need best predictive performance
\item \textcolor{mlcolor}{\textbf{Large, complex datasets}} - High-dimensional or non-linear patterns
\item \textcolor{mlcolor}{\textbf{Production systems}} - Computational resources available
\item \textcolor{mlcolor}{\textbf{Pattern discovery}} - Exploratory analysis on complex data
\item \textcolor{mlcolor}{\textbf{Non-linear relationships}} - When linear assumptions fail
\item \textcolor{dlcolor}{\textbf{Sequential/temporal data}} - LSTM/RNN for time series with complex patterns
\end{itemize}
\end{block}

\vspace{0.3cm}

\begin{center}
\textbf{Best Models:} Random Forest, SVM, LSTM/RNN, Gradient Boosting
\end{center}
\end{frame}

\begin{frame}{Key Takeaways}
\begin{enumerate}
    \item \textbf{No universal winner} - Both approaches have distinct advantages
    
    \vspace{0.3cm}
    
    \item \textbf{Statistical models excel in:}
    \begin{itemize}
        \item Interpretability (88.3/100 vs 37.4/100)
        \item Speed (20-100x faster training)
        \item Efficiency (6435.64 vs 0.36 efficiency scores)
        \item Regulatory compliance and hypothesis testing
    \end{itemize}
    
    \vspace{0.3cm}
    
    \item \textbf{ML/DL models excel in:}
    \begin{itemize}
        \item Complex dataset performance (0.818 vs 0.547 avg)
        \item Scalability to large data
        \item Capturing non-linear patterns
        \item Handling complex temporal dependencies
    \end{itemize}
    
    \vspace{0.3cm}
    
    \item \textbf{The optimal choice depends on:}
    \begin{itemize}
        \item Primary requirement (accuracy vs interpretability vs speed)
        \item Dataset characteristics (size, complexity, linearity)
        \item Computational constraints
        \item Regulatory requirements
    \end{itemize}
\end{enumerate}
\end{frame}
