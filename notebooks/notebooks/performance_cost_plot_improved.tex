% LaTeX TikZ Plot: Model Performance vs Training Cost

\begin{document}

\begin{tikzpicture}[

% Coordinate system:

% For time=2.5: log10(2.5) ≈ 0.4, x = (0.4 + 6) * 1.5 = 9.6

% Axes

% Grid (light gray)

\foreach \perf/\y in {0.2/0, 0.35/1.875, 0.5/3.75, 0.65/5.625, 0.8/7.5, 0.95/9.375}
    \node[left, font=\footnotesize] at (-0.3, \y) {\perf};

\node[above right=0.15cm of OLS, label, anchor=west] {OLS\\Regression\\\tiny (0.0001s, 0.65)};
